package architect

import (
    "fmt"
    "math"
    "time"
    "sync"
)

type DigitalArchitect struct {
    Name            string
    Username        string
    Age             int
    Status          string
    Skills          []string
    Passion         []string
    Mission         string
    ViewerCount     int64
    LastViewed      time.Time
    ViewerMutex     sync.RWMutex
}

func NewArchitect() *DigitalArchitect {
    architect := &DigitalArchitect{
        Name:        "X7R3O7 TIck",
        Username:    "X7R3O7-TIck",
        Age:         20,
        Status:      "🔴 LIVE - Crafting Digital Magic",
        Skills:      []string{"AI Architect", "Full-Stack Wizard", "Code Maestro", "Innovation Pioneer"},
        Passion:     []string{"Revolutionary Code", "AI Magic", "Digital Art", "Guitar Virtuoso", "Gaming Master"},
        Mission:     "Architecting the future, one innovative solution at a time",
        ViewerCount: 0,
        LastViewed:  time.Now(),
    }
    
    return architect
}

func (a *DigitalArchitect) IncrementViewers() {
    a.ViewerMutex.Lock()
    defer a.ViewerMutex.Unlock()
    a.ViewerCount++
    a.LastViewed = time.Now()
}

func (a *DigitalArchitect) GetLiveStats() {
    for {
        a.IncrementViewers()
        fmt.Printf("🔥 LIVE VIEWERS: %d | Last Visit: %s 🔥\n", 
                   a.ViewerCount, a.LastViewed.Format("15:04:05"))
        time.Sleep(2 * time.Second)
    }
}

func (a *DigitalArchitect) DisplayProfile() {
    fmt.Printf("⚡ DIGITAL ARCHITECT: %s ⚡\n", a.Name)
    fmt.Printf("🌟 Status: %s 🌟\n", a.Status)
    fmt.Printf("🚀 Mission: %s 🚀\n", a.Mission)
    fmt.Println("👨‍💻 THE CODE ARCHITECT IS ONLINE 👨‍💻")
}

func main() {
    architect := NewArchitect()
    architect.DisplayProfile()
    
    // Start live viewer tracking
    go architect.GetLiveStats()
    
    for {
        fmt.Println("🌟 Building the future with code... 🌟")
        time.Sleep(1 * time.Second)
    }
}
